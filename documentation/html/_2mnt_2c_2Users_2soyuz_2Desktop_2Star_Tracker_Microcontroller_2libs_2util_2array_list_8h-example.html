A resizable array which can be declared on the stack. A stack array which has list like properties. This has a fixed size. In default conditions, this is treated to copy the value when inserting. This stops the requirement of new in a local scope.

Template Parameters
TThe datatype to store.
NThe maximum size.
    ArrayList<int, 10> array(5);
    array.PushBack(123);
    cout << array.MaxSize() << endl;    // 10
    cout << array.Size() << endl;       // 6;
    cout << array.Get(5) << endl;       // 123;
    cout << array.PopBack() << endl;    // 123;
    array.ReduceSize(3);
    cout << array.Size() << endl;       // 3;

#pragma once
namespace util
{
template <class T, const int N>
class ArrayList
{
protected:
T array[N];
uint end;
public:
ArrayList ( )
{
end = 0;
}
ArrayList ( int size )
{
end = (size > N ? N : size);
}
ArrayList ( ArrayList& list )
{
end = 0;
for ( uint i = 0; i < list.Size(); i++ )
{
PushBack(list.Get(i));
}
}
virtual /*inline*/ bool IsEmpty ( ) const
{
return end == 0;
}
virtual /*inline*/ bool IsFull ( ) const
{
return end == N;
}
/*inline*/ constexpr int MaxSize ( )
{
return N;
}
virtual /*inline*/ uint Size ( ) const
{
return end;
}
void ReduceSize ( uint size )
{
if (end > size) end = size;
}
virtual T& Get ( uint position )
{
return array[position];
}
/*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*\
| |
| ---- Push/Pop ---- |
| |
\*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
virtual bool PushBack ( T val )
{
bool valid = !IsFull();
if ( valid )
{
array[end] = val;
end++;
}
return valid;
}
virtual T PopBack ( )
{
if ( !IsEmpty() )
{
end--;
return array[end];
}
return array[end];
}
/*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*\
| |
| ---- Others ---- |
| |
\*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*-*/
void Sort ( bool (*inOrder)(T&, T&) )
{
for ( uint ii = 1; ii < Size(); ii++ )
{
uint jj = ii;
T temp = array[jj];
while ( jj > 0 && inOrder(temp, array[jj - 1]) )
{
array[jj] = array[jj - 1];
jj--;
}
array[jj] = temp;
}
}
};
}
