154  void SpreadGrassFire ( uint threshold, Image* img )
155  {
156  util::ArrayList<util::Point<util::uint>, N> q;
157 
158  uint minX = round(centroid.x), minY = round(centroid.y);
159  uint maxX = minX, maxY = minY;
160 
161  q.PushBack(Point<uint>(minX, minY));
162 
163  while ( !q.IsEmpty() )
164  {
165  util::Point<util::uint> pt = q.PopBack();
166  if(img->ValidPixel(pt.x, pt.y)&&img->GetPixel(pt.x, pt.y)>threshold)
167  { // 4 directional
168 
169  q.PushBack(util::Point<uint>(pt.x + 1, pt.y));
170  q.PushBack(util::Point<uint>(pt.x - 1, pt.y));
171  q.PushBack(util::Point<uint>(pt.x, pt.y + 1));
172  q.PushBack(util::Point<uint>(pt.x, pt.y - 1));
173  // Diagonals
174  q.PushBack(util::Point<uint>(pt.x - 1, pt.y - 1));
175  q.PushBack(util::Point<uint>(pt.x - 1, pt.y + 1));
176  q.PushBack(util::Point<uint>(pt.x + 1, pt.y - 1));
177  q.PushBack(util::Point<uint>(pt.x + 1, pt.y + 1));
178 
179 
180  minX = (pt.x < minX ? pt.x : minX);
181  minY = (pt.y < minY ? pt.y : minY);
182  maxX = (pt.x > maxX ? pt.x : maxX);
183  maxY = (pt.y > maxY ? pt.y : maxY);
184 
185  const byte px = img->GetPixel(pt.x, pt.y);
186  // Setting new centroid.
187  centroid.x = FindCentroid(centroid.x, intensity, pt.x, px);
188  centroid.y = FindCentroid(centroid.y, intensity, pt.y, px);
189  // Setting intensity.
190  intensity += (uint) px;
191  // Setting pixels
192  pixels++;
193  // Stops reading same px
194  img->SetPixel(pt.x, pt.y, 0);
195  }
196  }
197  if ( pixels == 0 ) width = height = 0;
198  else
199  {
200  width = maxX - minX + 1;
201  height = maxY - minY + 1;
202  }
203  }
